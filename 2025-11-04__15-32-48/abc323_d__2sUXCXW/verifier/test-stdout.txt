def solve():
    n = int(input())
    slimes = {}

    for _ in range(n):
        s, c = map(int, input().split())
        slimes[s] = c

    # Process merges
    while True:
        # Find a size with count >= 2
        found = False
        for size in sorted(slimes.keys()):
            if slimes[size] >= 2:
                found = True
                # Merge two slimes of this size into one of double size
                count = slimes[size]
                remainder = count % 2
                new_count = count // 2

                # Update current size
                if remainder == 0:
                    del slimes[size]
                else:
                    slimes[size] = remainder

                # Add to double size
                new_size = size * 2
                if new_size in slimes:
                    slimes[new_size] += new_count
                else:
                    slimes[new_size] = new_count

                break

        if not found:
            break

    # Sum remaining slimes
    result = sum(slimes.values())
    print(result)

solve()

import sys

def main():
    N, Q = map(int, input().split())
    A = list(map(int, input().split()))

    for _ in range(Q):
        i, x = map(int, input().split())
        A[i - 1] = x  # Convert to 0-indexed

        # Calculate MEX
        # MEX is at most N, so we only check values from 0 to N
        A_set = set(A)
        mex = 0
        while mex in A_set:
            mex += 1

        print(mex)

if __name__ == "__main__":
    main()

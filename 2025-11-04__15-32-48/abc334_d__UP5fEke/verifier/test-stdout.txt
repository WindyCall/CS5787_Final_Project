# Sort the requirements in ascending order
reindeer_requirements.sort()

# For each query, find the maximum number of sleighs that can be pulled
for _ in range(q):
    x = int(input())

    count = 0
    total = 0

    for requirement in reindeer_requirements:
        if total + requirement <= x:
            total += requirement
            count += 1
        else:
            break

    print(count)

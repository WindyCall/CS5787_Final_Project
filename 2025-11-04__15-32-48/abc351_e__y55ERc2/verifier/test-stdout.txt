def calculate_distance(x1, y1, x2, y2):
    """Calculate minimum jumps from (x1, y1) to (x2, y2)"""
    dx = abs(x2 - x1)
    dy = abs(y2 - y1)

    # Each jump changes both x and y by Â±1
    # To reach (x2, y2) from (x1, y1), we need:
    # - max(dx, dy) jumps minimum
    # - If dx and dy have the same parity, it takes exactly max(dx, dy) jumps
    # - If they have different parity, we need at least max(dx, dy) more jumps
    #   but actually we need to reach a point where both differences are even

    # Let's think differently: after k jumps, if we go to (x1+ax, y1+ay)
    # then we've used jumps: (a_right + a_left) = k_x in x-direction
    # and (b_up + b_down) = k_y in y-direction
    # but also a_right - a_left = dx and b_up - b_down = dy
    # and we need one jump per move in each direction

    # Actually: each jump moves us diagonally, so in k jumps:
    # - we move in x: can be from -k to +k but must have same parity as k
    # - we move in y: can be from -k to +k but must have same parity as k

    # So we need: k >= max(dx, dy) and dx % 2 == k % 2 and dy % 2 == k % 2
    # This means dx % 2 == dy % 2

    if dx % 2 != dy % 2:
        # Different parity - impossible to reach
        return 0

    return max(dx, dy)

n = int(input())
points = []
for _ in range(n):
    x, y = map(int, input().split())
    points.append((x, y))

total = 0
for i in range(n):
    for j in range(i + 1, n):
        x1, y1 = points[i]
        x2, y2 = points[j]
        dist = calculate_distance(x1, y1, x2, y2)
        total += dist

print(total)

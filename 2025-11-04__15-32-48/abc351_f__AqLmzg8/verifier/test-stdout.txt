#!/usr/bin/env python3

def solve():
    n = int(input())
    a = list(map(int, input().split()))

    total = 0
    # For each pair (i, j) where i < j
    # Add max(A_j - A_i, 0)
    for i in range(n):
        for j in range(i + 1, n):
            total += max(a[j] - a[i], 0)

    print(total)

solve()

def solve():
    N, T = map(int, input().split())
    A = list(map(int, input().split()))

    # Create a grid with marked cells
    marked = [[False] * N for _ in range(N)]

    # Track counts for rows, columns, and diagonals
    row_count = [0] * N
    col_count = [0] * N
    diag1_count = 0  # top-left to bottom-right
    diag2_count = 0  # top-right to bottom-left

    # Process each announced number
    for turn in range(1, T + 1):
        num = A[turn - 1]

        # Convert number to row and column (1-indexed to 0-indexed)
        # Number is arranged as: 1 2 3 ... N, N+1 N+2 ... 2N, etc.
        row = (num - 1) // N
        col = (num - 1) % N

        # Mark the cell
        marked[row][col] = True
        row_count[row] += 1
        col_count[col] += 1

        # Check diagonals
        if row == col:
            diag1_count += 1
        if row + col == N - 1:
            diag2_count += 1

        # Check if Bingo is achieved
        # Check row
        if row_count[row] == N:
            print(turn)
            return

        # Check column
        if col_count[col] == N:
            print(turn)
            return

        # Check diagonal 1 (top-left to bottom-right)
        if diag1_count == N:
            print(turn)
            return

        # Check diagonal 2 (top-right to bottom-left)
        if diag2_count == N:
            print(turn)
            return

    # No Bingo achieved
    print(-1)

solve()

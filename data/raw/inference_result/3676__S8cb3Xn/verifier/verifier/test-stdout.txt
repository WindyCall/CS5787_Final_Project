class Solution:
    def smallestNumber(self, n: int) -> int:
        # Find the number of bits needed to represent n
        bits = n.bit_length()

        # Create a number with all bits set: 2^bits - 1
        candidate = (1 << bits) - 1

        # If candidate >= n, return it
        if candidate >= n:
            return candidate

        # Otherwise, we need one more bit: 2^(bits+1) - 1
        return (1 << (bits + 1)) - 1

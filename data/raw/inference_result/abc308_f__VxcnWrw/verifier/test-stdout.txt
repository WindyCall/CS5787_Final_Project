def solve():
    N, M = map(int, input().split())
    prices = list(map(int, input().split()))
    L = list(map(int, input().split()))
    D = list(map(int, input().split()))

    # Create coupons list: (discount, min_price)
    coupons = list(zip(D, L))

    # Sort items in descending order by price
    prices.sort(reverse=True)

    # Sort coupons: first by ASCENDING min_price (restrictive first),
    # then by DESCENDING discount (better coupons first within same restriction)
    coupons.sort(key=lambda x: (-x[1], -x[0]))

    # Track which items have been assigned a coupon
    item_assigned = [False] * N
    total_cost = sum(prices)  # Start with full cost

    # For each coupon (from most restrictive to least restrictive)
    for discount, min_price in coupons:
        # Find the best unassigned item that can use this coupon
        best_item_idx = -1
        best_item_price = -1

        for i in range(N):
            if not item_assigned[i] and prices[i] >= min_price:
                if prices[i] > best_item_price:
                    best_item_price = prices[i]
                    best_item_idx = i

        if best_item_idx != -1:
            # Assign this coupon to this item
            item_assigned[best_item_idx] = True
            total_cost -= discount

    print(total_cost)

solve()

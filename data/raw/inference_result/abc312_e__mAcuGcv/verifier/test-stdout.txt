def main():
    n = int(input())
    cuboids = []

    for _ in range(n):
        coords = list(map(int, input().split()))
        x1, y1, z1, x2, y2, z2 = coords
        cuboids.append((x1, y1, z1, x2, y2, z2))

    # For each cuboid, count how many others it shares a face with
    for i in range(n):
        count = 0
        for j in range(n):
            if i == j:
                continue

            if shares_face(cuboids[i], cuboids[j]):
                count += 1

        print(count)

def shares_face(cuboid1, cuboid2):
    """Check if two cuboids share a face (intersection has positive area)."""
    x1_min, y1_min, z1_min, x1_max, y1_max, z1_max = cuboid1
    x2_min, y2_min, z2_min, x2_max, y2_max, z2_max = cuboid2

    # Check each of the 6 faces of cuboid1
    # Top face (z = z1_max)
    if z1_max == z2_min and shares_rectangular_overlap(
        x1_min, y1_min, x1_max, y1_max,
        x2_min, y2_min, x2_max, y2_max
    ):
        return True

    # Bottom face (z = z1_min)
    if z1_min == z2_max and shares_rectangular_overlap(
        x1_min, y1_min, x1_max, y1_max,
        x2_min, y2_min, x2_max, y2_max
    ):
        return True

    # Front face (y = y1_max)
    if y1_max == y2_min and shares_rectangular_overlap(
        x1_min, z1_min, x1_max, z1_max,
        x2_min, z2_min, x2_max, z2_max
    ):
        return True

    # Back face (y = y1_min)
    if y1_min == y2_max and shares_rectangular_overlap(
        x1_min, z1_min, x1_max, z1_max,
        x2_min, z2_min, x2_max, z2_max
    ):
        return True

    # Right face (x = x1_max)
    if x1_max == x2_min and shares_rectangular_overlap(
        y1_min, z1_min, y1_max, z1_max,
        y2_min, z2_min, y2_max, z2_max
    ):
        return True

    # Left face (x = x1_min)
    if x1_min == x2_max and shares_rectangular_overlap(
        y1_min, z1_min, y1_max, z1_max,
        y2_min, z2_min, y2_max, z2_max
    ):
        return True

    return False

def shares_rectangular_overlap(a_min, b_min, a_max, b_max, c_min, d_min, c_max, d_max):
    """
    Check if two rectangles have a positive area overlap.
    Rectangle 1: [a_min, a_max] x [b_min, b_max]
    Rectangle 2: [c_min, c_max] x [d_min, d_max]
    """
    # Find intersection
    inter_a_min = max(a_min, c_min)
    inter_a_max = min(a_max, c_max)
    inter_b_min = max(b_min, d_min)
    inter_b_max = min(b_max, d_max)

    # Check if intersection has positive area
    return inter_a_min < inter_a_max and inter_b_min < inter_b_max

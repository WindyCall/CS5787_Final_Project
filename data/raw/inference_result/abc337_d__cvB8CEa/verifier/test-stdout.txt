def solve():
    # Read input
    H, W, K = map(int, input().split())
    grid = []
    for _ in range(H):
        grid.append(input().strip())

    min_ops = float('inf')

    # Check all horizontal lines of length K
    for i in range(H):
        for j in range(W - K + 1):
            # Count dots and check for x's in this horizontal line
            dots = 0
            has_x = False
            for k in range(K):
                if grid[i][j + k] == '.':
                    dots += 1
                elif grid[i][j + k] == 'x':
                    has_x = True

            # If no 'x' in this line, we can make it all 'o's
            if not has_x:
                min_ops = min(min_ops, dots)

    # Check all vertical lines of length K
    for i in range(H - K + 1):
        for j in range(W):
            # Count dots and check for x's in this vertical line
            dots = 0
            has_x = False
            for k in range(K):
                if grid[i + k][j] == '.':
                    dots += 1
                elif grid[i + k][j] == 'x':
                    has_x = True

            # If no 'x' in this line, we can make it all 'o's
            if not has_x:
                min_ops = min(min_ops, dots)

    # If no valid line found, return -1
    if min_ops == float('inf'):
        print(-1)
    else:
        print(min_ops)

solve()

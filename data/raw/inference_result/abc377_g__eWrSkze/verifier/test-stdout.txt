def lcp_length(s1, s2):
    """Calculate the length of the longest common prefix."""
    length = 0
    for i in range(min(len(s1), len(s2))):
        if s1[i] == s2[i]:
            length += 1
        else:
            break
    return length

def min_cost_to_transform(current, target):
    """Calculate minimum cost to transform current string to target."""
    lcp = lcp_length(current, target)
    return len(current) + len(target) - 2 * lcp

def solve():
    n = int(input())
    strings = []
    for _ in range(n):
        strings.append(input().strip())

    for k in range(n):
        current = strings[k]

        # Cost to make it empty
        min_cost = len(current)

        # Cost to match each previous string
        for i in range(k):
            cost = min_cost_to_transform(current, strings[i])
            min_cost = min(min_cost, cost)

        print(min_cost)

solve()

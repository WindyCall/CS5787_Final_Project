import math

def find_cubes(N):
    # x^3 - y^3 = (x-y)(x^2 + xy + y^2) = N
    # Let a = x - y, b = x^2 + xy + y^2
    # Then a * b = N

    # From x - y = a, we get x = y + a
    # Substituting: (y+a)^2 + (y+a)*y + y^2 = b
    # y^2 + 2ay + a^2 + y^2 + ay + y^2 = b
    # 3y^2 + 3ay + a^2 = b
    # 3y^2 + 3ay + (a^2 - b) = 0
    # y = (-3a ± sqrt(9a^2 - 12(a^2-b))) / 6
    # y = (-3a ± sqrt(12b - 3a^2)) / 6

    # Find all divisors of N
    divisors = []
    i = 1
    while i * i <= N:
        if N % i == 0:
            divisors.append(i)
            if i != N // i:
                divisors.append(N // i)
        i += 1

    # Try each divisor as 'a' (x - y)
    for a in divisors:
        b = N // a

        # Check if 12b - 3a^2 is non-negative
        discriminant = 12 * b - 3 * a * a

        if discriminant < 0:
            continue

        # Check if it's a perfect square
        sqrt_discriminant = int(math.sqrt(discriminant))
        if sqrt_discriminant * sqrt_discriminant != discriminant:
            continue

        # Try both cases
        for sign in [1, -1]:
            numerator = -3 * a + sign * sqrt_discriminant
            if numerator % 6 != 0:
                continue

            y = numerator // 6

            if y <= 0:
                continue

            x = y + a

            if x <= 0:
                continue

            # Verify the solution
            if x * x * x - y * y * y == N:
                return (x, y)

    return None

N = int(input())
result = find_cubes(N)
if result:
    print(result[0], result[1])
else:
    print(-1)
